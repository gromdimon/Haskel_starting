{-

Ergebnis:                   Teilen x durch y
Funktionssignatur:          teilen :: Float -> Float -> Float
Definintion:                teilen x y = x/y
-}


enthaeltZiffer :: Int -> Int -> Bool
enthaeltZiffer z n
    | 0 <= n && n <= 9 = n==z
    | otherwise = z == n`mod`10 || enthaeltZiffer z (n`div`10)
